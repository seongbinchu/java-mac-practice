DROP TABLE car;

CREATE TABLE car(
	car_id					INT					PRIMARY KEY		AUTO_INCREMENT,
	car_model				CHAR(200)			NOT NULL,
	car_price				INT					NOT	NULL,
	car_desc					VARCHAR(200)
);

-- table schema 확인
DESC car;

-- 가격이 2500 이상 , 5000이하의 차 구하라

SELECT car_id,car_model,car_price FROM car;
SELECT * FROM car WHERE car_price >= 2500 AND car_price =<5000;
SELECT * FROM car WHERE car_price BETWEEN 2500 AND 5000;

--가격이 3000이거나 3700인 자동차
SELECT * FROM car WHERE car_price =3000 OR car_price=3700;
SELECT * FROM car WHERE car_price IN(3000,3700);

--가격이 3000이 아닌 차량
SELECT * FROM car WHERE NOT car_price 3000;
SELECT * FROM car WHERE car_price <>3000;
SELECT * FROM car WHERE car_price !=3000;

SELECT * FROM car WHERE car_desc IS NULL;
-- FROM 절 1st = > WHERE => SELECT

--가격으로 내림차순 정렬
SELECT * FROM car ORDER BY car_price DESC;


--가격으로 오름차순 정렬하고 가격이 같으면 id로 내림차순 정렬
SELECT * FROM car ORDER by car_price,car_id DESC;

--가격이 2500 이상인 자동차를 오름차순
SELECT * FROM car where car_price>2500 ORDER BY car_price;
-- FROM => WHERE => SELECT => ORDER BY

--가격이 비싼 순으로 2~4등까지
SELECT * FROM car ORDER by car_price DESC LIMIT 1,3; 

-- wild card( % , _ )
-- % : 0 ~ 무한대 
-- _ : 글자 한개 

-- a로 시작하는 모든 모델명
SELECT * FROM car WHERE car_model LIKE 'a%';

-- a가 들어가는 모든 모델
SELECT * FROM car WHERE car_model LIKE '%a%';

-- 시작이 a인 6글자 생략안됨
SELECT * FROM car WHERE car_model LIKE 'a_____';

SELECT car_model,car_price, 
	CASE	
		WHEN car_price < 2000 THEN 'd' 
		WHEN car_price < 3000 THEN 'c'

		ELSE 'a'
	END AS grade
FROM car;
