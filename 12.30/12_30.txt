	public Vector<Car> getAll(Connection con){
		Vector<Car> list = new Vector<Car>();
		Statement stmt = null;
		
		//SELECT 문의 결과 담는 객체 table같이 생김
		ResultSet rs = null;
		String sql = "SELECT * FROM car";
		try {
			stmt = con.createStatement();
			rs = stmt.executeQuery(sql);
			while(rs.next()) {
				// rs.next  => table 헤더에서 시작[car_id...] 다음 줄이 존재하는가 ? => true => 한줄 내려감[커서이동]
				int car_id = rs.getInt("car_id");
				String car_model = toKor(rs.getString("car_model"));
				int car_price = rs.getInt("car_price");
				String car_desc = toKor(rs.getString("car_desc"));
				
				Car car = new Car(car_id,car_model,car_price,car_desc);
				list.add(car);
			}
		}catch(SQLException e) {
			e.printStackTrace();
		}finally {
			close(rs);
			close(stmt);
		}
		return list;
	}
	
	private void close(Statement stmt) {
		try {
			stmt.close();
		}catch(Exception e) {}
	}
	
	private void close(ResultSet rs) {
		try {
			rs.close();
		}catch(Exception e) {}
	}
	
	public Car findById(Connection con,int car_id) {
		Car car = new Car();
		Statement stmt = null;
		
		String sql="SELECT * FROM car WHERE car_id ="+car_id;
		ResultSet rs = null;
		try {
			stmt = con.createStatement();
			rs = stmt.executeQuery(sql);
			if(rs.next()) {
				car = rowMapping(rs);
			}
		}catch(SQLException e) {
			e.printStackTrace();
		}finally {
			close(rs);
			close(stmt);
		}
		return car;
	}
	
	private Car rowMapping(ResultSet rs) throws SQLException{
		int car_id = rs.getInt("car_id");
		String car_model = rs.getString("car_model");
		int car_price = rs.getInt("car_price");
		String car_desc = rs.getString("car_desc");
		
		Car car = new Car(car_id,car_model,car_price,car_desc);
		return car;
	}
	// INSERT,UPDATE : EUC-KR -> 8859_1
	
	private String toEn(String kor) {
		String en = null;
		try {
			en = new String(kor.getBytes("EUC_KR"),"8859_1");
		}catch(Exception e) {}
		return en;
	}
	// SELECT : 8859_1 => EUC-KR
	private String toKor(String en) {
		String kor = null;
		try {
			kor = new String(en.getBytes("8859_1"),"EUC-KR");
		}catch(Exception e) {}
		return kor;
	}
