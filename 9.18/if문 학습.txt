clas Quiz{
	public static void main(String args[]){
		int num;

		boolean flag = true;

		if(flag) {
		num = 5;
		}

		System.out.println(num);	

	}
}

//flag 와 true의 차이

//변수 와 상수의 차이

//변할 수 있고 변할 수 없음

//flag는 출력에 오류가 생기고 true 는 출력에 오류가 없음

//flag는 변수임으로 참일 때 5 거짓일 때 값이 없음으로 
//값을 지정할 수 없어 정의가 된 것으로 보지않는다

//ture의 경우 상수로써 변할 수 없기 때문에 num=5; 라는 고정 값을 가지고
//출력이 가능함

//else문으로 num에 하나의 값을 더 준다면 참 혹은 거짓이라는 두가지의
//경우수에서 둘다 값을 가지기 때문에 값이 출력됨 => 수행이 보장된 블럭

//if + else if(!flag)문의 조합은 수행을 보장하지 않는다 => 출력오류
//참 / !참(거짓) => 이성적으로 두가지의 경우의 수를 둘다 가지기 때문에
//경우의 수를 100% 만족시킬 수 있지만 기계는 그렇게 보지않음(조건이 변수)
